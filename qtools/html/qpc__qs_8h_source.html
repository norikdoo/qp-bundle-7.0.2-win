540#define QS_BEGIN_NOCRIT(rec_, qs_id_) \
541 if (QS_GLB_CHECK_(rec_) && QS_LOC_CHECK_(qs_id_)) { \
542 QS_beginRec_((uint_fast8_t)(rec_)); \
543 QS_TIME_PRE_(); {
544
546#define QS_END_NOCRIT() } \
547 QS_endRec_(); \
548 }
549
550#ifndef QS_REC_DONE
552 #define QS_REC_DONE() ((void)0)
553#endif /* QS_REC_DONE */
554
556#define QS_GLB_CHECK_(rec_) \
557 (((uint_fast8_t)QS_priv_.glbFilter[(uint_fast8_t)(rec_) >> 3U] \
558 & ((uint_fast8_t)1U << ((uint_fast8_t)(rec_) & 7U))) != 0U)
559
561#define QS_LOC_CHECK_(qs_id_) \
562 (((uint_fast8_t)QS_priv_.locFilter[(uint_fast8_t)(qs_id_) >> 3U] \
563 & ((uint_fast8_t)1U << ((uint_fast8_t)(qs_id_) & 7U))) != 0U)
564
574#define QS_BEGIN_ID(rec_, qs_id_) \
575 if (QS_GLB_CHECK_(rec_) && QS_LOC_CHECK_(qs_id_)) { \
576 QS_CRIT_STAT_ \
577 QS_CRIT_E_(); \
578 QS_beginRec_((uint_fast8_t)(rec_)); \
579 QS_TIME_PRE_(); {
580
585#define QS_END() } \
586 QS_endRec_(); \
587 QS_CRIT_X_(); \
588 }
